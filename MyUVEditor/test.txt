float4x4 WorldViewProj;
void VertexShader_Main(
	in float4 inPos		: POSITION,
	in float4 inDiff	: COLOR0,
	out float4 outPos	: POSITION,
	out float4 outDiff	: COLOR0)
{
	outPos = mul(inPos,WorldViewProj);
	outDiff = inDiff;
}

void PixelShader_Main(
	in float4 inDiff	: COLOR0,
	out float4 outDiff	: COLOR0)
{
	outDiff = inDiff;
}
